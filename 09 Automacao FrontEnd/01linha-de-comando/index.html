Linha de Comando

Comandos:

  ~ //entra no diretorio inicial (meu computador)
  ls //mostra os arquivos do local
  ls -all //mostra todos os arquivos detalhados inclusive os ocultos
  cd nomeDaPasta //serve para viaja até uma pasta
  cd Desktop //entra no na área de trabalho (desktop)
  cd .. //viaja para uma pasta anterior
  touch //cria um arquivo (.js, .html, .css, etc...)
  mkdir nomeDaPasta //cria uma pasta
  rm nomeDoArquivo //deleta um arquivo
  rm -r nomeDaPasta //deleta uma pasta
  clear //limpa a linha de Comando
